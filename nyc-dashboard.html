duckdb · nyc_trips · sample queries
DuckDB SQL
-- Total revenue and trips
SELECT COUNT(*) AS total_trips, SUM(total_amount) AS total_revenue FROM trips;

-- Peak hour analysis
SELECT HOUR(tpep_pickup_datetime) AS hour, COUNT(*) AS trips
FROM trips GROUP BY hour ORDER BY trips DESC;

-- Payment distribution
SELECT payment_type, COUNT(*) AS trips, ROUND(COUNT(*)*100.0/SUM(COUNT(*)) OVER(),1) AS pct
FROM trips GROUP BY payment_type ORDER BY trips DESC;
