athena · imdb_pipeline.movies · sample queries
Amazon Athena SQL
-- Panel 1: Top 10 Highest Rated Movies
SELECT primarytitle AS "Movie", averagerating AS "Rating", CAST(numvotes AS INTEGER) AS "Votes"
FROM imdb_pipeline.movies WHERE numvotes > 100000 ORDER BY averagerating DESC LIMIT 10;

-- Panel 2: Movies Per Decade
SELECT CONCAT(SUBSTRING(startyear,1,3),'0s') AS "Decade", COUNT(*) AS "Movies", ROUND(AVG(averagerating),2) AS "Avg Rating"
FROM imdb_pipeline.movies WHERE startyear IS NOT NULL GROUP BY SUBSTRING(startyear,1,3) ORDER BY "Decade";

-- Panel 3: Top Genres
SELECT genres AS "Genre", COUNT(*) AS "Movies" FROM imdb_pipeline.movies
WHERE genres != '\N' AND genres NOT LIKE '%,%' GROUP BY genres ORDER BY "Movies" DESC LIMIT 10;
